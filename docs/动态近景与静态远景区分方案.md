# 动态近景与静态远景区分方案

> **目标**：在代码层面实现近景/远景的差异化管理，优化游戏后期性能。
>
> **核心思想**：近景保持高精度动态更新，远景使用低精度静态缓存。

---

## 一、设计原则

### 1.1 为什么需要区分近景和远景？

| 区域 | 特征 | 更新频率 | 精度要求 |
|------|------|----------|----------|
| **近景** | 玩家可交互、可观察细节 | 高（实时） | 高（1:1） |
| **远景** | 仅提供视觉背景、不可交互 | 低（静态缓存） | 低（LOD 降采样） |

### 1.2 核心优化点

1. **更新策略差异**：近景区块在玩家移动时实时加载/卸载，远景区块只在首次加载时生成，之后保持静态
2. **网格精度差异**：近景使用全精度网格，远景使用 LOD 降采样网格
3. **内存管理差异**：近景区块数据保留在内存中用于交互，远景区块可只保留 GPU 网格
4. **脏标记差异**：近景区块支持 DirtyChunk 重建，远景区块忽略脏标记

```rust
// src/chunk_zone.rs

/// 区块区域类型：决定更新策略和渲染精度
#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
pub enum ChunkZone {
    /// 近景区域：玩家可交互范围
    /// - 实时加载/卸载
    /// - 全精度网格（LOD0）
    /// - 支持方块交互和脏标记重建
    /// - 保留完整区块数据在内存中
    Near,
    
    /// 远景区域：视觉背景范围
    /// - 首次加载后保持静态
    /// - LOD 降采样网格
    /// - 不支持交互
    /// - 可只保留 GPU 网格，释放区块数据
    Far,
    
    /// 缓冲区域：近景和远景之间的过渡带
    /// - 用于防止边界闪烁
    /// - 使用近景策略但优先级较低
    Buffer,
}

impl ChunkZone {
    /// 近景范围（区块数）：玩家周围此范围内的区块使用近景策略
    pub const NEAR_DISTANCE: i32 = 4;
    
    /// 远景起始范围（区块数）：超过此距离的区块使用远景策略
    pub const FAR_DISTANCE: i32 = 6;
    
    /// 根据距离计算区块区域
    pub fn from_distance(distance: i32) -> Self {
        match distance {
            0..=Self::NEAR_DISTANCE => ChunkZone::Near,
            (Self::NEAR_DISTANCE + 1)..=Self::FAR_DISTANCE => ChunkZone::Buffer,
            _ => ChunkZone::Far,
        }
    }
    
    /// 是否需要实时更新
    pub fn needs_realtime_update(&self) -> bool {
        matches!(self, ChunkZone::Near | ChunkZone::Buffer)
    }
    
    /// 是否支持方块交互
    pub fn supports_interaction(&self) -> bool {
        matches!(self, ChunkZone::Near)
    }
    
    /// 是否需要保留区块数据在内存中
    pub fn keeps_chunk_data(&self) -> bool {
        matches!(self, ChunkZone::Near)
    }
}
```

### 2.2 LOD 级别定义

```rust
// src/lod.rs

/// LOD（Level of Detail）级别
#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, PartialOrd, Ord)]
pub enum LodLevel {
    /// LOD0：全精度（1:1），用于近景
    Lod0 = 0,
    /// LOD1：1:2 降采样，用于中远景
    Lod1 = 1,
    /// LOD2：1:4 降采样，用于远景
    Lod2 = 2,
    /// LOD3：1:8 降采样，用于超远景
    Lod3 = 3,
}

impl LodLevel {
    /// 根据区块区域获取默认 LOD 级别
    pub fn from_zone(zone: ChunkZone) -> Self {
        match zone {
            ChunkZone::Near => LodLevel::Lod0,
            ChunkZone::Buffer => LodLevel::Lod1,
            ChunkZone::Far => LodLevel::Lod2,
        }
    }
    
    /// 降采样步长
    pub fn step(&self) -> usize {
        match self {
            LodLevel::Lod0 => 1,
            LodLevel::Lod1 => 2,
            LodLevel::Lod2 => 4,
            LodLevel::Lod3 => 8,
        }
    }
    
    /// 面数减少比例（相对于 LOD0）
    pub fn face_reduction_ratio(&self) -> f32 {
        match self {
            LodLevel::Lod0 => 1.0,
            LodLevel::Lod1 => 0.25,  // 75% 减少
            LodLevel::Lod2 => 0.0625, // 93.75% 减少
            LodLevel::Lod3 => 0.0156, // 98.4% 减少
        }
    }
}
```

### 3.2 区域计算系统

```rust
// src/chunk_zone.rs

/// 计算区块坐标的区域类型
pub fn calculate_chunk_zone(
    chunk_coord: ChunkCoord,
    player_chunk: ChunkCoord,
) -> ChunkZone {
    let dx = (chunk_coord.cx - player_chunk.cx).abs();
    let dz = (chunk_coord.cz - player_chunk.cz).abs();
    let distance = dx.max(dz); // 使用切比雪夫距离（棋盘距离）
    
    ChunkZone::from_distance(distance)
}

/// 更新所有区块的区域状态
pub fn update_chunk_zones(
    mut loaded: ResMut<LoadedChunks>,
    camera_query: Query<&Transform, With<Camera3d>>,
) {
    let Ok(cam_transform) = camera_query.single() else {
        return;
    };
    
    let player_chunk = ChunkCoord::from_world(cam_transform.translation);
    
    for (coord, entry) in loaded.entries.iter_mut() {
        let new_zone = calculate_chunk_zone(*coord, player_chunk);
        
        // 区域发生变化时的处理
        if entry.zone != new_zone {
            // 从近景切换到远景：释放区块数据，标记为静态
            if entry.zone == ChunkZone::Near && new_zone == ChunkZone::Far {
                if !new_zone.keeps_chunk_data() {
                    // 释放区块数据（只保留 GPU 网格）
                    entry.data = ChunkData::Empty; // 或使用 Option<ChunkData>
                }
                entry.is_static = true;
            }
            
            // 从远景切换到近景：需要重新加载区块数据
            if entry.zone == ChunkZone::Far && new_zone == ChunkZone::Near {
                entry.is_static = false;
                // 标记需要重新加载区块数据和重建网格
                // 这里需要触发异步任务重新生成
            }
            
            entry.zone = new_zone;
            entry.lod = LodLevel::from_zone(new_zone);
        }
    }
}
```

### 3.3 差异化更新策略

```rust
// src/chunk_manager.rs

/// 修改 chunk_loader_system 以支持区域差异化
pub fn chunk_loader_system(
    mut commands: Commands,
    mut materials: ResMut<Assets<VoxelMaterial>>,
    mut meshes: ResMut<Assets<Mesh>>,
    mut loaded: ResMut<LoadedChunks>,
    async_mesh: Res<AsyncMeshManager>,
    camera_query: Query<&Transform, With<Camera3d>>,
    atlas_handle: Res<AtlasTextureHandle>,
    uv_table: Res<UvLookupTable>,
    dirty_query: Query<&DirtyChunk>,
) {
    let Ok(cam_transform) = camera_query.single() else {
        return;
    };

    let player_chunk = ChunkCoord::from_world(cam_transform.translation);
    loaded.frame_counter += 1;
    let current_frame = loaded.frame_counter;

    // ── 步骤 1：收集异步结果 ──────────────────────────
    let results = async_mesh.collect_results(MESH_UPLOADS_PER_FRAME);
    for result in results {
        if !loaded.entries.contains_key(&result.coord) {
            continue;
        }

        if let Some(entry) = loaded.entries.get(&result.coord) {
            // 关键修改：静态区块跳过脏标记检查
            if entry.is_static {
                // 静态区块不响应脏标记，直接丢弃
                continue;
            }
            
            if dirty_query.get(entry.entity).is_ok() {
                continue;
            }
            
            // ... 上传 GPU 逻辑不变 ...
        }
    }

    // ── 步骤 2：检测玩家移动 ──────────────────────
    if loaded.last_player_chunk != Some(player_chunk) {
        loaded.last_player_chunk = Some(player_chunk);
        
        // 更新所有区块的区域状态
        update_chunk_zones_system(&mut loaded, player_chunk);
        
        rebuild_load_queue(player_chunk, &mut *loaded);
        unload_distant_chunks(/* ... */);
    }

    // ── 步骤 3：分帧提交异步任务 ────────────────────────
    let drain_count = CHUNKS_PER_FRAME.min(loaded.load_queue.len());
    let chunks_to_load: Vec<ChunkCoord> = loaded.load_queue.drain(..drain_count).collect();

    for coord in chunks_to_load {
        let zone = calculate_chunk_zone(coord, player_chunk);
        let lod = LodLevel::from_zone(zone);
        
        // 根据区域和 LOD 生成不同的网格
        let chunk_data = generate_chunk_data(coord);
        
        if zone.keeps_chunk_data() {
            // 近景：保留完整区块数据
            loaded.entries.get_mut(&coord).unwrap().data = chunk_data.clone();
        }
        
        // 提交异步网格生成任务（包含 LOD 信息）
        let neighbors = collect_neighbors(coord, &loaded);
        async_mesh.submit_task(MeshTask::Generate {
            coord,
            data: chunk_data,
            neighbors,
            lod, // 新增：传递 LOD 级别
        });
    }
}
```

### 3.4 LOD 网格生成

```rust
// src/chunk.rs

/// 生成 LOD 网格（支持降采样）
pub fn generate_chunk_mesh_lod(
    chunk: &Chunk,
    resource_pack: &ResourcePackManager,
    neighbors: &ChunkNeighbors,
    lod: LodLevel,
) -> (Vec<[f32; 3]>, Vec<[f32; 2]>, Vec<[f32; 3]>, Vec<u32>) {
    // 全空气区块提前返回
    if matches!(chunk, ChunkData::Empty | ChunkData::Uniform(0)) {
        return (Vec::new(), Vec::new(), Vec::new(), Vec::new());
    }

    let step = lod.step();
    
    // LOD0 使用原始函数
    if step == 1 {
        return generate_chunk_mesh(chunk, resource_pack, neighbors);
    }

    // LOD 降采样：每隔 step 个方块采样一次
    let mut positions = Vec::new();
    let mut uvs = Vec::new();
    let mut normals = Vec::new();
    let mut indices = Vec::new();

    // 降采样后的区块尺寸
    let lod_size = CHUNK_SIZE / step;
    
    for z in (0..CHUNK_SIZE).step_by(step) {
        for y in (0..CHUNK_SIZE).step_by(step) {
            for x in (0..CHUNK_SIZE).step_by(step) {
                // 采样该区域的主导方块类型
                let block_id = sample_dominant_block(chunk, x, y, z, step);
                if block_id == 0 {
                    continue;
                }

                // 为降采样后的"大方块"生成面
                for (face_index, (face, offset)) in FACES.iter().cloned().enumerate() {
                    // LOD 面剔除：检查降采样后的邻居
                    if !is_face_visible_lod(chunk, x, y, z, &offset, face_index, neighbors, step) {
                        continue;
                    }

                    let base_index = positions.len() as u32;
                    let face_name = face.to_face_name();
                    let uv = resource_pack
                        .get_block_uv(block_id, face_name)
                        .unwrap_or((0.0, 1.0, 0.0, 1.0));

                    // 生成降采样尺寸的面（step x step 大小）
                    let (face_verts, face_uvs, face_normal) = 
                        face_quad_lod(x, y, z, face, uv, step);
                    
                    positions.extend(face_verts);
                    uvs.extend(face_uvs);
                    normals.extend([face_normal; 4]);
                    indices.extend([
                        base_index,
                        base_index + 2,
                        base_index + 1,
                        base_index,
                        base_index + 3,
                        base_index + 2,
                    ]);
                }
            }
        }
    }

    (positions, uvs, normals, indices)
}

/// 采样区域内的主导方块类型
fn sample_dominant_block(
    chunk: &Chunk,
    x: usize,
    y: usize,
    z: usize,
    step: usize,
) -> BlockId {
    // 简单策略：取中心点的方块
    // 高级策略：统计区域内最常见的非空气方块
    let cx = (x + step / 2).min(CHUNK_SIZE - 1);
    let cy = (y + step / 2).min(CHUNK_SIZE - 1);
    let cz = (z + step / 2).min(CHUNK_SIZE - 1);
    chunk.get(cx, cy, cz)
}

/// LOD 版本的面可见性检查
fn is_face_visible_lod(
    chunk: &Chunk,
    x: usize,
    y: usize,
    z: usize,
    face: &[i32; 3],
    face_index: usize,
    neighbors: &ChunkNeighbors,
    step: usize,
) -> bool {
    // 检查邻居位置的方块（考虑步长）
    let nx = x as i32 + face[0] * step as i32;
    let ny = y as i32 + face[1] * step as i32;
    let nz = z as i32 + face[2] * step as i32;

    let current_id = sample_dominant_block(chunk, x, y, z, step);

    // 邻居在区块边界内
    if nx >= 0 && ny >= 0 && nz >= 0
        && nx < CHUNK_SIZE as i32
        && ny < CHUNK_SIZE as i32
        && nz < CHUNK_SIZE as i32
    {
        let neighbor_id = sample_dominant_block(
            chunk,
            nx as usize,
            ny as usize,
            nz as usize,
            step,
        );
        return neighbor_id != current_id;
    }

    // 邻居在区块边界外
    true // 简化处理：边界面总是可见
}

/// LOD 版本的面生成（生成更大的面）
fn face_quad_lod(
    x: usize,
    y: usize,
    z: usize,
    face: Face,
    uv: (f32, f32, f32, f32),
    step: usize,
) -> ([[f32; 3]; 4], [[f32; 2]; 4], [f32; 3]) {
    let s = step as f32;
    let (verts, normal) = match face {
        Face::Top => (
            [
                [x as f32, y as f32 + s, z as f32],
                [x as f32 + s, y as f32 + s, z as f32],
                [x as f32 + s, y as f32 + s, z as f32 + s],
                [x as f32, y as f32 + s, z as f32 + s],
            ],
            [0.0, 1.0, 0.0],
        ),
        // ... 其他面类似 ...
        _ => todo!(),
    };

    let (u_min, u_max, v_min, v_max) = uv;
    let face_uvs = [
        [u_min, v_max],
        [u_max, v_max],
        [u_max, v_min],
        [u_min, v_min],
    ];

    (verts, face_uvs, normal)
}
```
